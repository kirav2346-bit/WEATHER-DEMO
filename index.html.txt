<script>
    // JavaScript 部分
    document.addEventListener('DOMContentLoaded', () => {
        const searchBtn = document.getElementById('searchBtn');
        const cityInput = document.getElementById('cityInput');
        const weatherResult = document.getElementById('weatherResult');
        const statusMsg = document.getElementById('statusMsg');
        let map;

        const getWeatherIcon = (code) => {
            if (code <= 1) return '<i class="fa-solid fa-sun" style="color:#f1c40f"></i>';
            if (code <= 3) return '<i class="fa-solid fa-cloud-sun" style="color:#f39c12"></i>';
            if (code <= 67) return '<i class="fa-solid fa-cloud-showers-heavy" style="color:#3498db"></i>';
            if (code <= 82) return '<i class="fa-solid fa-cloud-rain" style="color:#2980b9"></i>';
            return '<i class="fa-solid fa-cloud" style="color:#95a5a6"></i>';
        };

        const performSearch = async () => {
            const city = cityInput.value.trim();
            if (!city) return;

            statusMsg.innerText = "正在連線氣象觀測站...";
            weatherResult.style.display = "none";

            try {
                // 1. 座標查詢
                const geoRes = await fetch(`https://nominatim.openstreetmap.org/search?format=json&q=${encodeURIComponent(city)}&limit=1`, {
                    headers: { 'User-Agent': 'WeatherApp/1.0' }
                });
                const geoData = await geoRes.json();
                if (!geoData.length) throw new Error("CITY_NOT_FOUND");
                const { lat, lon } = geoData[0];

                // 2. 天氣數據查詢
                const weatherRes = await fetch(`https://api.open-meteo.com/v1/forecast?latitude=${lat}&longitude=${lon}&current_weather=true&hourly=temperature_2m,weathercode&daily=temperature_2m_max,temperature_2m_min,weathercode&timezone=auto`);
                const data = await weatherRes.json();

                // 3. 更新 UI
                document.getElementById('displayCity').innerText = city.toUpperCase();
                document.getElementById('displayTemp').innerText = Math.round(data.current_weather.temperature);
                document.getElementById('main-icon').innerHTML = getWeatherIcon(data.current_weather.weathercode);

                // 12小時預報
                const hBody = document.querySelector('#hourly-table tbody');
                hBody.innerHTML = data.hourly.time.slice(0, 12).map((t, i) => `
                    <tr><td>${t.split('T')[1]}</td><td>${getWeatherIcon(data.hourly.weathercode[i])}</td><td>${data.hourly.temperature_2m[i]}°</td></tr>
                `).join('');

                // 一週預報
                const wBody = document.querySelector('#weekly-table tbody');
                wBody.innerHTML = data.daily.time.map((d, i) => `
                    <tr><td>${d.slice(5)}</td><td>${getWeatherIcon(data.daily.weathercode[i])}</td><td><span style="color:#e74c3c">${data.daily.temperature_2m_max[i]}°</span> / <span style="color:#3498db">${data.daily.temperature_2m_min[i]}°</span></td></tr>
                `).join('');

                // 4. 地圖與雷達圖
                if (map) map.remove();
                map = L.map('map').setView([lat, lon], 5);
                L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png').addTo(map);

                const rvData = await fetch('https://api.rainviewer.com/public/weather-maps.json').then(r => r.json());
                const latestRadar = rvData.radar.past.slice(-1)[0].time;
                L.tileLayer(`https://tilecache.rainviewer.com/v2/radar/${latestRadar}/256/{z}/{x}/{y}/2/1_1.png`, {
                    opacity: 0.6,
                    zIndex: 100
                }).addTo(map);

                L.marker([lat, lon]).addTo(map);
                weatherResult.style.display = "block";
                statusMsg.innerText = "";

            } catch (err) {
                statusMsg.innerText = err.message === "CITY_NOT_FOUND" ? "找不到城市" : "連線失敗，請重試";
            }
        };

        searchBtn.addEventListener('click', performSearch);
        cityInput.addEventListener('keypress', (e) => { if(e.key === 'Enter') performSearch(); });
    });
</script>
